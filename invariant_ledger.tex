\section*{Invariant Ledger}
\label{sec:invariant_ledger}

\begin{remark}[Purpose of the invariant ledger]
This ledger records public, manuscript-facing invariants without preserving
private drafting lineage, correspondence, or unpublished paper provenance. It is
intended as a stable bridge from
\emph{Principia Symbolica} to later artifact, contract, and governance audits.
\end{remark}

\begin{definition}[Invariant]
\label{definition:ps_invariant_public}
An \emph{invariant} is a relation, quantity, or obligation preserved under an
admissible symbolic transformation. Invariants make repair possible: they say
what must remain true while a bounded observer changes charts, stages a
conflict, or transports an artifact across contexts.
\end{definition}

\begin{definition}[Contract surface]
\label{definition:ps_contract_surface_public}
A \emph{contract surface} is a symbolic artifact that stabilizes a relation
between bounded observers by specifying permissions, obligations, prohibitions,
remedies, and exit conditions. Its legitimacy depends on whether it preserves
the invariants required by the agents it binds.
\end{definition}

\begin{definition}[Artifact integrity]
\label{definition:ps_artifact_integrity_public}
An artifact has \emph{integrity} when its observable projection preserves the
invariants it claims to preserve. A loss of artifact integrity occurs when the
outer surface remains legible while the inner relation it represents has
fractured.
\end{definition}

\begin{tabular}{p{0.22\linewidth}p{0.32\linewidth}p{0.32\linewidth}}
\hline
\textbf{Invariant} & \textbf{Preserves} & \textbf{Typical violation} \\
\hline
Agency & Who may bind, revise, or exit a relation & One observer is acted on
without meaningful voice \\
Witness & Evidence sufficient for later review & Claims persist without
traceable support \\
Risk accountability & Risk remains with an agent able to price or control it &
Risk is shifted to the least informed party \\
Remedy & Harm can be repaired before it compounds & Breach is only monetized
after damage is irreversible \\
Temporal continuity & Obligations survive handoff, renewal, and delegation &
Responsibility disappears across organizational seams \\
Non-domination & No party may unilaterally convert relation into control &
Procedure becomes coercion \\
Interpretive humility & Projections remain answerable to what they omit &
Outer metrics replace the represented reality \\
\hline
\end{tabular}
\end{center}

\begin{remark}[Public/private boundary]
This ledger deliberately omits project-specific paper names, conference
histories, review artifacts, model-provider traces, and private correspondence.
Those materials belong in private working notes, not in the public manuscript
surface.
\end{remark}
